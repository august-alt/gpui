87 T readIntegralFromBuffer(std::istream &buffer)

92  if (buffer.fail()) {
93  if (buffer.eof()) {
94  throw std::runtime_error(
95  "LINE: " + std::to_string(__LINE__) + ", FILE: " + __FILE__

97  }
98  throw std::runtime_error(
99  "LINE: " + std::to_string(__LINE__) + ", FILE: " + __FILE__
100  + ", Failed to read integral number from buffer, error was encountered.");
101  }
102  if constexpr (LE) {
103  return leToNative<T>(num);
